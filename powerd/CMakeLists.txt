cmake_minimum_required( VERSION 3.12 )

set( POWERD_DIR ${PROJECT_BASE_DIR}/powerd )

include_directories( ${PROJECT_BASE_DIR}/utils )
include_directories( ${PROJECT_BASE_DIR}/common )
include_directories( ${PROJECT_BASE_DIR}/WTI )

link_directories( ${PROJECT_BASE_DIR}/lib )

add_executable(powerd 
        ${POWERD_DIR}/powerd.cpp 
        ${POWERD_DIR}/power_interface.cpp 
        )

target_link_libraries(powerd
        wti
        network
        common
        logentry
        utilities
        ${CMAKE_THREAD_LIBS_INIT}
        )

# -- Port Definitions ----------------------------------------------------------
#
# portdefs is the master list of ports for NGPS and is used to generate the
# individual .cfg files to avoid conflicts across multiple files, by keeping
# them all in one place..
#
INCLUDE( ${PROJECT_BASE_DIR}/Config/portdefs )
set ( POWERD_CFG_COMMENT "This file is automatically generated. Manual edits may be overwriten." )
CONFIGURE_FILE( ${PROJECT_BASE_DIR}/Config/powerd.cfg.in ${PROJECT_BASE_DIR}/Config/powerd.cfg )
