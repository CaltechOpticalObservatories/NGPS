# ----------------------------------------------------------------------------
# @file    telemd/CMakeLists.txt
# @brief   input to the CMake build system for building telemetry daemon
# @author  David Hale <dhale@caltech.edu>
# ----------------------------------------------------------------------------

cmake_minimum_required( VERSION 3.12 )

set( TELEMD_DIR ${PROJECT_BASE_DIR}/telemd )

set( CMAKE_CXX_STANDARD 17 )

add_definitions( -Wall -ansi -O1 -Wno-variadic-macros -ggdb )

include_directories( ${PROJECT_BASE_DIR}/utils )
include_directories( ${PROJECT_BASE_DIR}/common )

add_executable(telemd 
        ${TELEMD_DIR}/telemd.cpp 
        ${TELEMD_DIR}/telem_server.cpp 
        ${TELEMD_DIR}/telem_interface.cpp 
        )

target_link_libraries(telemd
        network
        common
        logentry
        utilities
        ${CMAKE_THREAD_LIBS_INIT}
        )

# -- Port Definitions ----------------------------------------------------------
#
# portdefs is the master list of ports for NGPS and is used to generate the
# individual .cfg files to avoid conflicts across multiple files, by keeping
# them all in one place..
#
INCLUDE( ${PROJECT_BASE_DIR}/Config/portdefs )
set ( TELEMD_CFG_COMMENT "This file is automatically generated. Manual edits may be overwritten." )
CONFIGURE_FILE( ${PROJECT_BASE_DIR}/Config/telemd.cfg.in ${PROJECT_BASE_DIR}/Config/telemd.cfg )
